# Lab 2: Initializing the SAAYN Workspace

## Objective

In this lab, you will initialize SAAYN’s local workspace and inspect how this repository already uses chunk markers and a chunk registry. In this version of the project, the source files are already marked with SAAYN chunk boundaries, so this lab focuses on preparing the workspace and examining the structures SAAYN relies on.

* Running `saayn init` to prepare local SAAYN directories and configuration
* Inspecting existing `SAAYN:CHUNK_START` and `SAAYN:CHUNK_END` markers in the source
* Exploring `chunk-registry.json` to see how SAAYN tracks chunks
* Verifying which commands are actually available in this version of the tool

   ```go
   package cmd

   import (
           "encoding/json"
           "fmt"
           "os"
           "path/filepath"

           "saayn/internal/registry"

           "github.com/spf13/cobra"
   )
   // SAAYN:CHUNK_END:init-imports-v1-a1b2c3d4

   // SAAYN:CHUNK_START:init-command-definition-v1-e5f6g7h8
   var initCmd = &cobra.Command{
           Use:   "init",
           Short: "Initialize SAAYN in the current project",
           Long: `Bootstraps the .saayn internal directory, the chunk registry,
   and environment template for a sovereign project.`,
           Run: func(cmd *cobra.Command, args []string) {
                   runInit()
           },
   }

   func init() {
           rootCmd.AddCommand(initCmd)
   }
   // SAAYN:CHUNK_END:init-command-definition-v1-e5f6g7h8

   // SAAYN:CHUNK_START:init-logic-v1-i9j0k1l2
   func runInit() {
           fmt.Println("🏗️  Initializing SAAYN project...")

           // 1. Create .saayn internal directory structure
           paths := []string{
                   ".saayn/backup",
                   ".saayn/journal",
           }
           for _, p := range paths {
                   if err := os.MkdirAll(p, 0700); err != nil {
                           fmt.Printf("❌ Failed to create directory %s: %v\n", p, err)
                           return
                   }
           }

           // 2. Initialize Empty Registry if not exists
           registryPath := "chunk-registry.json"
           if _, err := os.Stat(registryPath); os.IsNotExist(err) {
                   emptyRegistry := registry.Registry{Chunks: []registry.Chunk{}}
                   data, _ := json.MarshalIndent(emptyRegistry, "", "  ")
                   if err := os.WriteFile(registryPath, data, 0644); err != nil {
                           fmt.Printf("❌ Failed to create registry: %v\n", err)
                           return
                   }
                   fmt.Println("📄 Created chunk-registry.json")
           }

           // 3. Setup .env template (12-Factor Config)
           envPath := ".env"
           if _, err := os.Stat(envPath); os.IsNotExist(err) {
                   template := `# SAAYN Configuration
   SAAYN_INFERENCE_URL="http://localhost:8000/v1"
   SAAYN_PLANNER_MODEL="llama-3-8b"
   SAAYN_CODER_MODEL="qwen-2.5-coder-32b"
   SAAYN_AUTO_COMMIT=false
   `
                   os.WriteFile(envPath, []byte(template), 0600)
                   fmt.Println("🔐 Created .env template (ensure you add your keys/endpoints)")
           }

           // 4. Safety: Ensure .env and .saayn/ are in .gitignore
           ensureGitIgnore()

           fmt.Println("✅ SAAYN initialization complete. Start adding chunks to your files!")
   }
   ```
